識別子


禁則文字
:= 　( 　　| '\'' | '"' 　　| '!' | '^' | '+' | '-' | '*' | '/' | '%' 　　| ':' | '=' | '&' | '#' | '@' | '~' | '|' | '\\' 　　| '.' | ',' | '?' | '`' 　　| '<' | '>' | '[' | ']' | '{' | '}' 　　) ;
識別子
:= 　( ( 任意文字 - 禁則文字 - ';' - 空白 ) ( 任意文字 - 禁則文字 - ';' - 空白 )* ) - 予約語 - 定数
;

完全修飾識別子
:= 　識別子 ( '.' 完全修飾識別子 )?
;
